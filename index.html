Usage

    <!-- Import Bootstrap and jQuery, and this JS -->
    <script src="js/jquery.animated-progressbar.js"></script>

    <!-- Declare your progressbar -->
    <div class="progress">
        <div class="bar" data-progressbar="on" 
                         data-progressbar-begin="0"
                         data-progressbar-end="1" 
                         data-progressbar-delay="100" 
                         data-progressbar-duration="10000" >
        </div>
    </div>

    <!-- activate the progressbar -->
    <script type="text/javascript">
        $(".bar[data-progressbar='on']").each(function() {
            //animated_progressbar($(this));
			$(this).animatedProgressbar();
        });
    </script>
